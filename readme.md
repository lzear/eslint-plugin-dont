The ultimate ESLint plugin.

### www.dont.ninja

```js
     // Make your code look like this!
         import 👈 from 'left-pad'
         import 🎭 from 'is-even'

               let ✋ = '1'
            let 🔥 = +👈(✋++, ✋)
       let 🌈 = 🔥++ + ++✋ + +🎭(++🔥)
            let 🌯 = 🌈 + 🔥 * 🖐️

// 12% of the people can't solve this! 😂
       console.log(👈(✋, 🌯, 🌈) + 1)
```
